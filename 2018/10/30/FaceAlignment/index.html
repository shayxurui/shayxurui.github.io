###根据高斯核，生成高斯图:
image=np.zeros((256,256))
point=landmark
sigma=2

ul = [math.floor(point[0] - 3 * sigma), math.floor(point[1] - 3 * sigma)]     #ul=[xmin-6,ymin-6]
br = [math.floor(point[0] + 3 * sigma), math.floor(point[1] + 3 * sigma)]     #br=[xmax-6,ymax+6]
if (ul[0] > image.shape[1] or ul[1] >
        image.shape[0] or br[0] < 1 or br[1] < 1):
    return image

size = 6 * sigma + 1
g = _gaussian(size)  #高斯核  13*13
g_x = [int(max(1, -ul[0])), int(min(br[0], image.shape[1])) -int(max(1, ul[0])) + int(max(1, -ul[0]))]
g_y = [int(max(1, -ul[1])), int(min(br[1], image.shape[0])) -int(max(1, ul[1])) + int(max(1, -ul[1]))]
img_x = [int(max(1, ul[0])), int(min(br[0], image.shape[1]))]
img_y = [int(max(1, ul[1])), int(min(br[1], image.shape[0]))]
assert (g_x[0] > 0 and g_y[1] > 0)
image[img_y[0] - 1:img_y[1], img_x[0] - 1:img_x[1]] = image[img_y[0] - 1:img_y[1], img_x[0] - 1:img_x[1]] + g[g_y[0] - 1:g_y[1], g_x[0] - 1:g_x[1]]
#image[ymin-1:ymax,xmin-1:xmax]=image[ymin-1:ymax,xmin-1:xmax]+g[1-1:y_max-ymin+1,1-1:x_max-xmin+1]
image[image > 1] = 1
return image
